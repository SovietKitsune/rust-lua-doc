-- Highlighter and type information
local tippystring = require('tealdoc.tippy')
local themerecord = require('tealdoc.theme')
local lexer<invalid type> = require('pl.lexer')
local tl<invalid type> = require('tl')

local record Tokentype record
   --- A basic type, it will attempt to query `parser.createType` to generate the basic type
   typ: tl<invalid type>.TypeInfo -- parser.Type
   --- The kind of token it is
   kind: string
   --- The contents of what the token is
   str: string
   line: integer
end

local function castKindfunction(string, string): string(kindstring: string, textstring: string): string
   if kindstring == 'identifier' and textstring == '_G' or textstring == '_ENV' then
      return 'constant'
   elseif kindstring == 'identifier' and (_G as {string: any})[textstring] then
      return 'library'
   elseif kindstring == 'identifier' and textstring == 'self' then
      return 'keyword2'
   else
      return kindstring
   end
end

--; TODO - merge parse and fullDocument into a single function?
-- It would be faster theoretically
local function parsefunction(string, function(string): string, FILE, {string}): {Token}, tl.TypeReport(filestring: string, searchfunction(string): string, FILE, {string}: function(module: string): string, FILE, {string}): {Tokentype record}, tl<invalid type>.TypeReport
   local fFILE = io.open(filestring)

   local contentsstring = fFILE:read('*a')

   fFILE:close()

   local oldSearch<invalid type> = tl<invalid type>.search_module

   -- The new search isolates context which is needed to prevent external modules
   tl<invalid type>.search_module = searchfunction(string): string, FILE, {string}

   local result<invalid type> = tl<invalid type>.process_string(contentsstring, false, nil, filestring)
   local report<invalid type> = tl<invalid type>.get_types(result<invalid type>)

   tl<invalid type>.search_module = oldSearch<invalid type>

   local tokens{Token} = {} as {Tokentype record}

   local stream<invalid type> = lexer<invalid type>.lua(contentsstring, {
      space = false,
      comments = false
   }, {
      number = false,
      string = false
   })

   local xinteger = 1
   local yinteger = 1

   for kind, text in stream<invalid type> do
      kind = castKindfunction(string, string): string(kind, text)

      local lineinteger = stream<invalid type>(0) as integer

      if yinteger ~= lineinteger then
         yinteger = lineinteger
         xinteger = 1
      end

      xinteger = xinteger + #text

      local types<invalid type> = tl<invalid type>.symbols_in_scope(report<invalid type>, lineinteger, xinteger)

      local indexinteger: integer

      for i<invalid type>, v<invalid type> in pairs(types<invalid type>) do
         if i<invalid type> == text then
            indexinteger = v<invalid type>
            break
         end
      end

      local tokenToken = {
         kind = kind,
         str = text,
         lineinteger = lineinteger,
         typ = report<invalid type>.types<invalid type>[indexinteger]
      } as Tokentype record

      table.insert(tokens{Token}, tokenToken)
   end

   return tokens{Token}
end

local entitiesrecord = {
   ['<'] = '&lt;',
   ['>'] = '&gt;',
   ['&'] = '&amp;'
}

local function escapefunction(string): string(textstring: string): string
   return textstring:gsub('[<>&]', entitiesrecord) as string
end

-- No teal involved here, just highlighting
local function fastHighlightfunction(string, boolean): string(textstring: string, noWrapboolean: boolean): string
   local body{} = {}

   local stream<invalid type> = lexer<invalid type>.lua(textstring, {
      space = false,
      comments = false
   }, {
      number = false,
      string = false
   })

   for kind, textstring in stream<invalid type> do
      kind = castKindfunction(string, string): string(kind, textstring)

      if kind == 'whitespace' then
         table.insert(body{}, textstring)
      elseif themerecord.isNonText(kind) then
         table.insert(body{}, '<span class="code-' .. kind .. '">' .. escapefunction(string): string(textstring) .. '</span>')
      else
         table.insert(body{}, '<span class="code-text">' .. escapefunction(string): string(textstring) .. '</span>')
      end
   end

   if noWrapboolean then
      return table.concat(body{})
   else
      return '<pre class="code-block"><code>' .. table.concat(body{}) .. '</code></pre>'
   end
end

local function fullDocumentfunction({Token}, string, string, boolean, string): string(tokens{Token}: {Tokentype record}, currentstring: string, selectedThemestring: string, noOutlineboolean: boolean, toRootstring: string): string
   toRootstring = toRootstring or '../'

   local body{} = {}

   local yinteger = 1

   for iinteger = 1, #tokens{Token} do
      local tokenToken = tokens{Token}[iinteger]
      local within{} = {}
      local tagstring = 'span'
      local extrastring = ''

      if tokenToken.kind == 'whitespace' then
         table.insert(body{}, tokenToken.str)
      else
         if tokenToken.line ~= yinteger then
            yinteger = tokenToken.line

            table.insert(body{}, '<span id="line-' .. yinteger .. '"></span>')
         end

         if tokenToken.typ then
            table.insert(within{}, 'data-tippy-content')

            extrastring = '<span style="display: none;"><code>' .. fastHighlightfunction(string, boolean): string(tokenToken.typ.str, true) .. '</code></span>'

            if tokenToken.typ.yinteger then
               tagstring = 'a'

               local src<invalid type> = tokenToken.typ.file

               local hrefstring: string

               if src<invalid type> ~= currentstring and src<invalid type> then
                  hrefstring = toRootstring .. 'sources/' .. src<invalid type>:gsub('/', '-') .. '.html'
               else
                  hrefstring = '#line-' .. tokenToken.typ.yinteger
               end

               table.insert(within{}, 'href="' .. hrefstring .. '"')
            end
         end

         local colorstring = themerecord.isNonText(tokenToken.kind) and tokenToken.kind or 'text'

         if tokenToken.str == 'true' or tokenToken.str == 'false' or tokenToken.str == 'nil' then
            colorstring = 'number'
         end

         table.insert(within{}, 'class="code-' .. colorstring .. '"')

         table.insert(body{}, '<' .. tagstring .. ' ' .. table.concat(within{}, ' ') .. '>' .. escapefunction(string): string(tokenToken.str) .. extrastring .. '</' .. tagstring .. '>')
      end
   end

   if noOutlineboolean then
      return table.concat(body{})
   end

   return ([[
   <!DOCTYPE html>
   <html lang="en">

   <head>
      <meta charset="UTF-8">
      <meta http-equiv="X-UA-Compatible" content="IE=edge">
      <meta name="viewport" content="width=device-width, initial-scale=1.0">
      <style>
         a { text-decoration: none; }
         %s
      </style>
   </head>
   <body class="%s">
      <pre class="code-block"><code>
%s
      </code></pre>
      %s
   </body>
   ]]):format(themerecord.css, selectedThemestring, table.concat(body{}), tippystring)
end

return {
   parsefunction(string, function(string): string, FILE, {string}): {Token}, tl.TypeReport = parsefunction(string, function(string): string, FILE, {string}): {Token}, tl.TypeReport,
   fullDocumentfunction({Token}, string, string, boolean, string): string = fullDocumentfunction({Token}, string, string, boolean, string): string,
   fastHighlightfunction(string, boolean): string = fastHighlightfunction(string, boolean): string
}
