--- The configuration passed to templates
local record Projecttype record
   name: string
   version: string
   deprecated: boolean | string
   description: string
   module: parserrecord.Type | nil
   modules: {string: parserrecord.Type} | nil
   sources: {string: string}
   functionStyle: string
   output: string
   theme: string
end

--- The user-provided configuration
local record Configtype record
   --- @{hidden}
   enum FunctionStyle
      'lua'
      'moon'
   end
   --- @{hidden}
   enum TagStyle
      'ldoc'
      'tealdoc'
      'none'
   end
   --- @{hidden}
   enum Theme
      'moonlight'
      'light'
   end
   --- The name of the project, if not provided this will be taken from a rockspec
   name: string
   --- The version of the project, if not provided this will be taken from a rockspec
   version: string
   --- If the project is deprecated and why it was deprecated
   deprecated: boolean | string
   --- The description of the project, if not provided this will be taken from:
   ---
   --- `descriptionFile -> rockspec.description.detailed -> rockspec.description.summary -> readme.md`
   description: string
   --- The file to use for the description
   descriptionFile: string
   --- If the project is written in Lua, defaults to false
   isLua: boolean
   --- The function style, defaults to `lua`
   ---
   --- `lua`:
   ---
   --- ```lua
   --- function foo(bar: baz): baz
   --- ```
   ---
   --- `moon`:
   ---
   --- ```moon
   --- foo = (bar: baz) -> baz
   --- ```
   functionStyle: FunctionStyle
   --- The output directory, defaults to `docs/`
   output: string
   --- Where to find all the project source files, defaults to current working directory
   source: string
   --- The tag format that should be used, this usually doesn't matter for Teal besides for hiding and deprecations
   ---
   --- The default is `tealdoc`
   ---
   --- `tealdoc`:
   ---
   --- Note that the `-` after `@` is only there to escape the tag
   ---
   --- ```lua
   --- --- This is a cool function
   --- -- @-{param y number}
   --- -- @-{return number}
   --- local function x(y)
   ---    return y
   --- end
   --- ```
   ---
   --- `ldoc`:
   ---
   --- ```lua
   --- --- This is a cool function
   --- -- @tparam number y
   --- -- @treturn number
   --- local function x(y)
   ---    return y
   --- end
   --- ```
   ---
   --- `none`:
   ---
   --- Doesn't preload any tags
   tagStyle: TagStyle
   --- The directories where extra tags are located
   tags: {string}
   --- The theme to use, the theme can be changed on the client side as all styles are loaded
   ---
   ---The theme defaults to `light`.
   ---
   --- `light`:
   ---
   --- This is like one-light but with some changed elements
   ---
   --- `moonlight`
   ---
   --- This is the "dark" theme inspired by <https://github.com/atomiks/moonlight-vscode-theme>
   theme: Theme
end

local record ConfigRockspectype record
   record Description
      summary: string
      detailed: string
   end
   package: string
   version: string
   description: Description
end

local function dirTreefunction(string, {string : string}, string): {string : string}(dirstring: string, current{string : string}: {string: string}, depthstring: string): {string: string}
   local tree{string : string} = current{string : string} or {}

   depthstring = (depthstring or '') .. (dirstring:match('.*/(.*)$') or dirstring)

   if depthstring ~= '' then
      depthstring = depthstring .. '.'
   end

   depthstring = depthstring:match('%.*(.*)')

   for path in lfs<invalid type>.dirstring(dirstring) do
      if path ~= '.' and path ~= '..' then
         local attributes<invalid type> = lfs<invalid type>.attributes<invalid type>(dirstring .. '/' .. path)

         if attributes<invalid type>.mode == 'file' and (path:match('%.lua$') or path:match('%.tl$')) then
            tree{string : string}[depthstring .. path:match('(.-)%.(.*)')] = dirstring .. '/' .. path
         elseif attributes<invalid type>.mode == 'directory' then
            dirTreefunction(string, {string : string}, string): {string : string}(dirstring .. '/' .. path, tree{string : string}, depthstring)
         end
      end
   end

   return tree{string : string}
end

local function searchfunction(string, {string}): string(patstring: string, dirs{string}: {string}): string
   for iinteger = 1, #dirs{string} do
      if lfs<invalid type>.attributes(dirs{string}[iinteger]) and lfs<invalid type>.attributes(dirs{string}[iinteger]).mode == 'directory' then
         for path in lfs<invalid type>.dir(dirs{string}[iinteger]) do
            if path:match(patstring) then
               return path
            end
         end
      end
   end
end

local function recursiveDirfunction(string, string | nil, {string}): {string}(pathstring: string, patstring | nil: string | nil, current{string}: {string}): {string}
   current{string} = current{string} or {}

   for location in lfs<invalid type>.dir(pathstring) do
      if location ~= '.' and location ~= '..' then

         if lfs<invalid type>.attributes(pathstring .. '/' .. location).mode == 'directory' then
            recursiveDirfunction(string, string | nil, {string}): {string}(pathstring .. '/' .. location, patstring | nil, current{string})
         elseif lfs<invalid type>.attributes(pathstring .. '/' .. location).mode == 'file' and location:match(patstring | nil) then
            table.insert(current{string}, pathstring .. '/' .. location)
         end
      end
   end

   return current{string}
end

local function currentLocationfunction(): string(): string
   local strstring = debug.getinfo(2, 'S').source:sub(2)

   -- The CLI is **never** ran directly from the interpreter

   return strstring:match('(.*/)')
end

local function readfunction(string): string(pathstring: string): string
   local fFILE = assert(io.open(pathstring, 'r'))

   local datastring = fFILE:readfunction(string): string('*a')

   fFILE:close()

   return datastring
end

local function loadTagfunction(string): <any type>(filestring: string): any
   local datastring = readfunction(string): string(filestring)

   local fnfunction(<any type>): <any type> = assert(load(datastring, 'tag'))

   return fnfunction(<any type>): <any type>()
end

local function loadTagsfunction(Config): {tag.TagConfig}(configConfig: Configtype record): {tagrecord.TagConfig}
   local files{} = {}

   local defaultTagsstring = currentLocationfunction(): string() .. '/tags/'

   if configConfig.tagStyle == 'ldoc' then
      loggerrecord.warn('LDoc support hasn\'t been implemented yet!')
      recursiveDirfunction(string, string | nil, {string}): {string}(defaultTagsstring .. 'ldoc', '%.lua$', files{})
   elseif configConfig.tagStyle == 'tealdoc' then
      recursiveDirfunction(string, string | nil, {string}): {string}(defaultTagsstring .. 'tealdoc', '%.lua$', files{})
   elseif configConfig.tagStyle ~= 'none' then
      loggerrecord.error('Unrecognized tag style: ' .. configConfig.tagStyle)
   end

   configConfig.tags = configConfig.tags or {}

   for iinteger = 1, #configConfig.tags do
      recursiveDirfunction(string, string | nil, {string}): {string}(configConfig.tags[iinteger], '%.lua$', files{})
   end

   local tags{} = {}

   for iinteger = 1, #files{} do
      table.insert(tags{}, loadTagfunction(string): <any type>(files{}[iinteger]))
   end

   return tags{} as {tagrecord.TagConfig}
end

local function writefunction(string, string)(pathstring: string, datastring: string)
   local fFILE = assert(io.open(pathstring, 'w'))

   fFILE:writefunction(string, string)(datastring)

   fFILE:close()
end

local function loadSpecfunction(string): ConfigRockspec(specstring: string): ConfigRockspectype record
   local datastring = readfunction(string): string(specstring)

   local env{} = {}

   local fnfunction(<any type>): <any type> = assert(load(datastring, 'rockspec', 't', env{}))

   if tonumber(_VERSION:match('Lua ([%d%.]+)')) < 5.2 then
      global setfenv: function(function, table)

      -- Lua 5.1 compat
      setfenv(fnfunction(<any type>): <any type>, env{})
   end

   pcall(fnfunction(<any type>): <any type>)

   return env{} as ConfigRockspectype record
end

local function loadConfigfunction(string): Config(contentsstring: string): Configtype record
   -- Dynamic requires in teal
   local fnfunction(<any type>): <any type> = assert(load(contentsstring, 'config'))

   return fnfunction(<any type>): <any type>() as Configtype record
end

local argParser<invalid type> = argparse<invalid type>('tealdoc', 'A documentation generator for Teal and by extension Lua')

argParser<invalid type>:option('-c --config', 'The configuration file location', 'tealdoc.lua', io.open)

--- The entry-point of the CLI
---
--- Flags:
---
--- * `-c`/`--config` - The configuration file location
--- * `-q`/`--quiet` - Suppress output
local function mainfunction()()
   local parsed<invalid type> = argParser<invalid type>:parse()

   loggerrecord.level =
      (parsed<invalid type>.quiet and 1) or
      (os.getenv('DEBUG') and 5) or
      3

   loggerrecord.color = not os.getenv('NO_COLOR')

   local fileContentsstring = (parsed<invalid type>.config as FILE):readfunction(string): string('*a')

   local configConfig = loadConfigfunction(string): Config(fileContentsstring)

   local rockspecConfigRockspec: ConfigRockspectype record
   -- Fallback to rockspec
   local specstring = searchfunction(string, {string}): string('.*%.rockspec', {'.', 'rockspecs'})

   if specstring then
      rockspecConfigRockspec = loadSpecfunction(string): ConfigRockspec(specstring)
   end

   local projectProject = {} as Projecttype record

   projectProject.name = configConfig.name or rockspecConfigRockspec.package or error('Expected package to have a name')
   projectProject.version = configConfig.version or rockspecConfigRockspec.version or ''
   projectProject.description =
      configConfig.description or
      (configConfig.descriptionFile and readfunction(string): string(configConfig.descriptionFile)) or
      (rockspecConfigRockspec.description and rockspecConfigRockspec.description.detailed or rockspecConfigRockspec.description.summary) or
      (lfs<invalid type>.attributes('readme.md') and readfunction(string): string('readme.md')) or
      ''
   projectProject.deprecated = configConfig.deprecated
   projectProject.functionStyle = configConfig.functionStyle or 'lua'

   if not configConfig.source then
      configConfig.source = tostring(configConfig.name)
   end

   configConfig.output = configConfig.output or 'docs'
   configConfig.tagStyle = configConfig.tagStyle or 'tealdoc'

   projectProject.output = configConfig.output
   projectProject.theme = configConfig.theme or 'moonlight'

   if not configConfig.output then
      lfs<invalid type>.mkdir(configConfig.output)
   end

   assert(lfs<invalid type>.attributes(configConfig.source))

   local tree{string : string} = dirTreefunction(string, {string : string}, string): {string : string}(configConfig.source)

   local stateState = parserrecord.getReport(tree{string : string}, false, configConfig.isLua)

   projectProject.sources = stateState.files

   local mods{string : integer} = parserrecord.extractModules(stateState, tree{string : string})

   local types{} = {}
   local countinteger = 0
   local firststring: string

   for istring, vinteger in pairs(mods{string : integer}) do
      types{}[istring] = parserrecord.createType(vinteger, stateState)

      countinteger = countinteger + 1

      if not firststring then
         firststring = istring
      end
   end

   if countinteger == 1 then
      projectProject.module = types{}[firststring]
   else
      projectProject.modules = types{}
   end

   local modSearchfunction(string): string, FILE, {string} = parserrecord.newSearch(tree{string : string}, false, stateState)
   local writer<invalid type> = lunamark<invalid type>.writer<invalid type>.html.new({})

   local currentSourcestring: string

   -- Change some defaults since of tailwind messing with them

   writer<invalid type>.link = function(labelstring: string, uristring: string, titlestring: string): {string}
      local titleAttstring = ''

      if type(titlestring) == 'string' and #titlestring > 0 then
         titleAttstring = 'title="' .. titlestring .. '"'
      end

      return {'<a href="', uristring, '" class="text-blue-500" ', titleAttstring, '>', labelstring, '</a>'}
   end

   writer<invalid type>.header = function(strstring: string, levelinteger: integer): {string}
      local textMaps{string} = {
         '3xl',
         '2xl',
         'xl',
         'lg',
         'base',
         'base'
      }

      return {'<h', tostring(levelinteger), ' class="text-', textMaps{string}[levelinteger], ' pt-2">', strstring, '</h', tostring(levelinteger), '>'}
   end

   -- We don't want anything else
   writer<invalid type>.template = '$body'

   writer<invalid type>.paragraph = function(strstring: string): {string}
      return {'<p class="py-2">', strstring, '</p>'}
   end

   writer<invalid type>.fenced_code = function(codestring: string, languagestring: string): {string} --;ignore
      if languagestring == 'teal' then
         local tmpstring = os.tmpname()

         -- *virtual*
         local fFILE = io.open(tmpstring, 'w')
         fFILE:writefunction(string, string)(codestring)
         fFILE:close()

         local rootstring: string

         if not currentSourcestring then
            rootstring = './'
         else
            local currentSlashesinteger = 0

            for iinteger = 1, #currentSourcestring do
               if currentSourcestring:sub(iinteger, iinteger) == '/' then
                  currentSlashesinteger = currentSlashesinteger + 1
               end
            end

            rootstring = './' .. string.rep('../', currentSlashes)
         end

         local tokens{Token} = highlightrecord.parse(tmpstring, modSearchfunction(string): string, FILE, {string})

         local contentsstring = highlightrecord.fullDocument(tokens{Token}, tmpstring, projectProject.theme, true, rootstring)

         return {'<pre class="code-block"><code>', contentsstring, '</code></pre>'}
      elseif languagestring == 'lua' then
         return {highlightrecord.fastHighlight(codestring)}
      else
         return {'<pre class="code-block"><code>', codestring, '</code></pre>'}
      end
   end

   local markdown<invalid type> = lunamark<invalid type>.reader.markdown<invalid type>.new(writer<invalid type>, {
      fenced_code_blocks = true --;ignore
   })

   local scriptDirstring = currentLocationfunction(): string()
   local cache{string : function(<any type>): string}: {string: function(any): string} = {}

   local function includefunction(string, {<any type> : <any type>}): string(filenamestring: string, base{<any type> : <any type>}: table): string
      base{<any type> : <any type>} = base{<any type> : <any type>} or {}

      base{<any type> : <any type>}.projectProject = projectProject
      base{<any type> : <any type>}.includefunction(string, {<any type> : <any type>}): string = function(fstring: string, b{<any type> : <any type>}: table): string
         b{<any type> : <any type>} = b{<any type> : <any type>} or {}

         for i<any type>, v<any type> in pairs(base{<any type> : <any type>}) do
            b{<any type> : <any type>}[i<any type>] = v<any type>
         end

         return includefunction(string, {<any type> : <any type>}): string(fstring, b{<any type> : <any type>})
      end

      cache{string : function(<any type>): string}[filenamestring] = cache{string : function(<any type>): string}[filenamestring] or assert(etlua<invalid type>.compile(readfunction(string): string(scriptDirstring .. 'templates/' .. filenamestring)))

      return cache{string : function(<any type>): string}[filenamestring](base{<any type> : <any type>})
   end

   -- Templates --
   local humanizefunction(parser.Type, boolean): string = require('tealdoc.humanize')(projectProject as table)

   local module<invalid type> = assert(etlua<invalid type>.compile(readfunction(string): string(scriptDirstring .. 'templates/module.etlua')))
   local document<invalid type> = assert(etlua<invalid type>.compile(readfunction(string): string(scriptDirstring .. 'templates/document.etlua')))

   local function genDocumentfunction(string, parser.Type)(namestring: string, tpparser.Type: parserrecord.Type)
      local locationstring: string

      -- I feel like this will lead to recursion
      tpparser.Type.tree{string : string} = tpparser.Type.parent and (tpparser.Type.parent as parserrecord.Type).tree{string : string} or ''

      tpparser.Type.namestring = namestring

      currentSourcestring = tpparser.Type.tree{string : string}

      if tpparser.Type.kind == 'custom' and not tpparser.Type.struct.isRecord then
         tpparser.Type.tree{string : string} = tpparser.Type.tree{string : string} .. '/' .. namestring

         currentSourcestring = tpparser.Type.tree{string : string}

         lfs<invalid type>.mkdir(configConfig.output .. '/' .. tpparser.Type.tree{string : string})

         locationstring = configConfig.output .. tpparser.Type.tree{string : string} .. '/index.html'

         writefunction(string, string)(locationstring, module<invalid type>({
            projectProject = projectProject,
            current = tpparser.Type,
            includefunction(string, {<any type> : <any type>}): string = includefunction(string, {<any type> : <any type>}): string,
            humanizefunction(parser.Type, boolean): string = humanizefunction(parser.Type, boolean): string,
            markdown<invalid type> = markdown<invalid type>,
            highlightrecord = highlightrecord.fastHighlight
         }))

         for istring, vType in pairs(tpparser.Type.struct.fields) do
            -- Generate children
            if not vType.hidden then
               genDocumentfunction(string, parser.Type)(istring, vType)
            end
         end
      else
         -- Non-modules get a document
         locationstring = configConfig.output .. tpparser.Type.tree{string : string} .. '/' .. namestring .. '.html'

         writefunction(string, string)(locationstring, document<invalid type>({
            projectProject = projectProject,
            current = tpparser.Type,
            humanizefunction(parser.Type, boolean): string = humanizefunction(parser.Type, boolean): string,
            markdown<invalid type> = markdown<invalid type>,
            includefunction(string, {<any type> : <any type>}): string = includefunction(string, {<any type> : <any type>}): string,
            highlightrecord = highlightrecord.fastHighlight
         }))
      end

      loggerrecord.info('Generated ' .. locationstring)
      currentLocationfunction(): string = nil
   end

   local tags{tag.TagConfig} = loadTagsfunction(Config): {tag.TagConfig}(configConfig)

   tagrecord.traverse(projectProject.modules or {
      [projectProject.name] = projectProject.module<invalid type> as parserrecord.Type
   }, tags{tag.TagConfig})

   writefunction(string, string)(configConfig.output .. '/index.html', module<invalid type>({
      current = projectProject,
      projectProject = projectProject,
      includefunction(string, {<any type> : <any type>}): string = includefunction(string, {<any type> : <any type>}): string,
      humanizefunction(parser.Type, boolean): string = humanizefunction(parser.Type, boolean): string,
      markdown<invalid type> = markdown<invalid type>
   }))

   loggerrecord.info('Generated ' .. configConfig.output .. '/index.html')

   if projectProject.module<invalid type> then
      assert(not (projectProject.module<invalid type> as parserrecord.Type).hidden, 'Hiding the entire project sounds a bit stupid doesn\'t it?')
      genDocumentfunction(string, parser.Type)(projectProject.name, projectProject.module<invalid type> as parserrecord.Type)
   else
      for istring, vparser.Type in pairs((projectProject.modules as {string: parserrecord.Type})) do
         if not vparser.Type.hidden then
            genDocumentfunction(string, parser.Type)(istring, vparser.Type)
         end
      end
   end

   -- Generate sources

   lfs<invalid type>.mkdir(configConfig.output .. '/' .. 'sources')

   for istring in pairs(projectProject.sources) do
      local tokens{Token} = highlightrecord.parse(istring, modSearchfunction(string): string, FILE, {string})

      local contentsstring = highlightrecord.fullDocument(tokens{Token}, istring, projectProject.theme)

      writefunction(string, string)(configConfig.output .. '/sources/' .. istring:gsub('/', '-') .. '.html', contentsstring)

      loggerrecord.info('Generated ' .. configConfig.output .. '/sources/' .. istring:gsub('/', '-') .. '.html')
   end
end

--- The main function of the documentation generator as well as the project config
return {
   Projecttype record = Projecttype record,
   Configtype record = Configtype record,
   mainfunction() = mainfunction()
}
