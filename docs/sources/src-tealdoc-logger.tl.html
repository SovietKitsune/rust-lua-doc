--- Logger for debugging and logging
local loggerrecord = {
   --- The current level of the logger
   level = 3,
   --- If there should be color in the output
   color = true
}

local human{string} = {
   'Error ',
   'Warn  ',
   'Info  ',
   'Debug ',
   'Trace '
}

local colors{integer} = {
   1,
   3,
   2,
   6,
   5
}

local function fgfunction(number): string(colornumber: number): string
   if not loggerrecord.colornumber then return '' end
   return '\27[38;5;' .. colornumber .. 'm'
end

local function resetfunction(): string(): string
   if not loggerrecord.color then return '' end

   return '\27[39m'
end

--- Create a log message, the level must be a number between 1 and 5
function loggerrecord.log(levelinteger: integer, messagestring: string)
   if levelinteger > loggerrecord.levelinteger then return end

   messagestring = messagestring:gsub('\n', '\n' .. fgfunction(number): string(colors{integer}[levelinteger]) .. human{string}[levelinteger] .. resetfunction(): string())

   local fileFILE = levelinteger >= 3 and io.stdout or io.stderr

   fileFILE:write(fgfunction(number): string(colors{integer}[levelinteger]) .. human{string}[levelinteger] .. resetfunction(): string() .. messagestring .. '\n')
end

--- Send an error level message
function loggerrecord.error(messagestring: string) loggerrecord.log(1, message) end
--- Send a warning level message
function loggerrecord.warn(messagestring: string) loggerrecord.log(2, message) end
--- Send an informational message
function loggerrecord.info(messagestring: string) loggerrecord.log(3, message) end

--- Send a debug message
---
--- The debug message will be inspected and colorized to aid in debugging (both `inspect.lua` and `pl.lexer` must be installed)
function loggerrecord.debug(...(<any type>): any)
   local messages{<any type>} = {...(<any type>)}

   local successboolean, inspect<invalid type> = pcall(require, 'inspect')

   if successboolean then
      for iinteger = 1, select('#', ...(<any type>)) do
         local message<any type> = select(iinteger, ...(<any type>))

         message<any type> = (inspect<invalid type> as function(any, table): string)(message<any type>, {
            indent = '   '
         })

         local hasLexerboolean, lexer<invalid type> = pcall(require, 'pl.lexer')

         if hasLexerboolean then
            local buffer{string}: {string} = {}

            for token, val in lexer<invalid type>.lua((message<any type> as string), {}, {}) do
               if token == 'comment' then
                  table.insert(buffer{string}, fgfunction(number): string(0))
               elseif token == 'keyword' and val ~= 'true' and val ~= 'false' then
                  table.insert(buffer{string}, fgfunction(number): string(5))
               elseif token == 'string' then
                  table.insert(buffer{string}, fgfunction(number): string(2))
               elseif token == 'number' or token == 'keyword' and (val == 'false' or val == 'true') then
                  table.insert(buffer{string}, fgfunction(number): string(3))
               elseif token == 'iden' and val == 'self' then
                  table.insert(buffer{string}, fgfunction(number): string(11))
               end

               table.insert(buffer{string}, val .. resetfunction(): string())
            end

            message<any type> = table.concat(buffer)
         end

         messages{<any type>}[iinteger] = (message<any type> as string)
      end
   end

   loggerrecord.log(4, table.concat(messages{<any type>} as {string}, '\t'))
end

--- Get the traceback where all build-file paths are replaced with source-file paths
function loggerrecord.trace()
   -- Not the cleanest but doesn't need to be
   local msgstring = debug.traceback()
      :gsub('\t', fgfunction(number): string(colors{integer}[5]) .. human{string}[5] .. resetfunction(): string())
      :gsub('%./out/', './src/')
      :gsub('%.lua:', '.tl:')

   loggerrecord.log(5, msgstring)
end

return loggerrecord
