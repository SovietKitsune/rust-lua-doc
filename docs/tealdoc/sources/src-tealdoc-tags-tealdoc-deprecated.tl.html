local rulesrecord = require('tealdoc.tag').rulesrecord
local parserrecord = require('tealdoc.parser')

--- @{hidden}
return {
   selector = '@{deprecated (.-)}',
   allowed = rulesrecord.any(
      rulesrecord.kind('function'),
      rulesrecord.parentMustBe('custom'),
      rulesrecord.parentMustBe('enum')
   ),
   fn = function(typparser.Type: parserrecord.Type, reasonstring: string)
      typparser.Type.deprecated = reasonstring or true
   end,
   err = 'The `deprecated` tag must be used on either a function or a child of a record/table/enum!'
}
