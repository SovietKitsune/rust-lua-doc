local parserrecord = require('tealdoc.parser')

local project{<any type> : <any type>}: table
local genFnfunction(parser.Type, boolean): function(fn: parserrecord.Type, style: boolean)

local primitive{parser.Type.Kinds : boolean}: {parserrecord.Type.Kinds: boolean} = {
    ['nil'] = true,
    ['boolean'] = true,
    ['number'] = true,
    ['integer'] = true,
    ['string'] = true,
    ['userdata'] = true,
    ['thread'] = true,
    ['any'] = true,
    ['unknown'] = true
}

local links{parser.Type.Kinds : string}: {parserrecord.Type.Kinds: string} = {
    ['nil'] = 'https://www.lua.org/pil/2.1.html',
    ['boolean'] = 'https://www.lua.org/pil/2.2.html',
    ['number'] = 'https://www.lua.org/pil/2.3.html',
    ['integer'] = 'https://www.lua.org/pil/2.3.html',
    ['string'] = 'https://www.lua.org/pil/2.4.html',
    ['table'] = 'https://www.lua.org/pil/2.5.html',
    ['function'] = 'https://www.lua.org/pil/2.6.html',
    ['userdata'] = 'https://www.lua.org/pil/2.7.html',
    ['thread'] = 'https://www.lua.org/pil/2.7.html',
    ['any'] = 'https://www.lua.org/pil/contents.html#2',
    ['unknown'] = 'https://www.lua.org/contents.html#2'
}

local function humanizefunction(parser.Type, boolean): string(typparser.Type: parserrecord.Type, styleboolean: boolean): string
    local kindKinds = typparser.Type.kindKinds

    if primitive{parser.Type.Kinds : boolean}[kindKinds] then
        return styleboolean and '<a href="' .. links{parser.Type.Kinds : string}[kindKinds] .. '" class="text-blue-500">' .. kindKinds .. '</a>' or kindKinds
    elseif kindKinds == 'named' then
        return typparser.Type.customName
    elseif kindKinds == 'function' then
        return genFnfunction(parser.Type, boolean)(typparser.Type, styleboolean)
    elseif kindKinds == 'table' then
        if (typparser.Type.key as parserrecord.Type).kindKinds == 'number' then
            return '{' .. humanizefunction(parser.Type, boolean): string(typparser.Type.value, styleboolean) .. '}'
        elseif (typparser.Type.key as parserrecord.Type).kindKinds == 'any' and (typparser.Type.value as parserrecord.Type).kindKinds == 'any' then
            return styleboolean and '<a href="' .. links{parser.Type.Kinds : string}[kindKinds] .. '" class="text-blue-500">' .. kindKinds .. '</a>' or kindKinds
        else
            return '{' .. humanizefunction(parser.Type, boolean): string(typparser.Type.key, styleboolean) .. ': ' .. humanizefunction(parser.Type, boolean): string(typparser.Type.value, styleboolean) .. '}'
        end
    elseif kindKinds == 'generic' then
        return styleboolean and '<span class="text-green-500">' .. typparser.Type.name .. '</span>' or typparser.Type.name
    elseif kindKinds == 'custom' or kindKinds == 'enum' then
        local namestring = typparser.Type.namestring
        return styleboolean and '<a href="./' .. namestring .. '.html" class="text-blue-500">' .. namestring .. '</a>' or name
    elseif kindKinds == 'union' or kindKinds == 'tuple' then
        local types{} = {}

        for inumber = 1, #typparser.Type.types{} do
            local humanizedstring = humanizefunction(parser.Type, boolean): string((typparser.Type.types{} as {parserrecord.Type})[inumber], styleboolean)
            table.insert(types{}, humanizedstring)
        end

        local resstring = table.concat(types{}, kindKinds == 'union' and ' | ' or ', ')

        if kindKinds == 'tuple' then
            return '(' .. resstring .. ')'
        else
            return resstring
        end
    end
end

genFnfunction(parser.Type, boolean) = function(tpparser.Type: parserrecord.Type, styleboolean: boolean): string
    local fnparser.Function = tpparser.Type.func as parserrecord.Function
    local fnNamestring = tpparser.Type.name or ''
    local srcstring: string

    if project{<any type> : <any type>}.functionStyle == 'moon' then
        srcstring = (styleboolean and '<b>' .. fnNamestring .. '</b>' or fnNamestring) .. (fnNamestring ~= '' and ' = ' or '')
    else
        srcstring = (styleboolean and '<b>function</b>' or 'function') .. (fnNamestring ~= '' and ' ' or '') .. fnNamestring
    end

    local parens{} = {}
    local generics{} = {}
    local rets{} = {}

    for inumber = 1, #fnparser.Function.params do
        local paramType = fnparser.Function.params[inumber]

        if paramType.kind == 'generic' then
            table.insert(generics{}, paramType.name)
        end

        local paramNamestring = (paramType.name and paramType.name .. ': ' or '') .. humanizefunction(parser.Type, boolean): string(paramType, styleboolean)

        table.insert(parens{}, paramNamestring)
    end

    for inumber = 1, #fnparser.Function.returns do
        local humanizedstring = humanizefunction(parser.Type, boolean): string(fnparser.Function.returns[inumber], styleboolean)
        table.insert(rets{}, humanizedstring)
    end

    local retStrstring: string

    local genericsStrstring = #generics{} > 0 and '<' .. table.concat(generics{}, ', ') .. '>' or ''
    local parensStrstring = '(' .. table.concat(parens{}, ', ') .. ')'

    if project{<any type> : <any type>}.functionStyle == 'moon' then
        retStrstring = #rets{} > 0 and ' -> ' .. table.concat(rets{}, ', ') or ''
    else
        retStrstring = #rets{} > 0 and ': ' .. table.concat(rets{}, ', ') or ''
    end

    srcstring = srcstring .. genericsStrstring .. parensStrstring .. retStrstring

    return srcstring
end

--- A factory used to generate a function that converts types into strings
return function(pro{<any type> : <any type>}: table): function(tp: parserrecord.Type, style: boolean): string
    project{<any type> : <any type>} = pro{<any type> : <any type>}

    return humanizefunction(parser.Type, boolean): string
end
