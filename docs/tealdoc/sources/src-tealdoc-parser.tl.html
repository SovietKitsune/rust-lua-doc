local loggerrecord = require('tealdoc.logger')
local tl<invalid type> = require('tl')

--- A structure that represents a bare function
local record Functiontype record
   params: {Type}
   returns: {Type}
   vararg: boolean
   isMetamethod: boolean
end

--- A structure that represents a bare record or table
local record Structtype record
   fields: {string: Type}
   isRecord: boolean
end

--- A structure that represents every type
local record Typetype record
   --- A list of all valid type names
   enum Kinds
      'nil'
      'boolean'
      'number'
      'integer'
      'string'
      'userdata'
      'thread'
      'table'
      --- Explicit type as any
      'any'
      --- A type which couldn't be figured out
      'unknown'
      --- Either an inline function or an actual function
      'function'
      --- A custom type like a record or module
      'custom'
      --- A union type that combines multiple types
      'union'
      --- A table which must contain a certain number of types in the correct order
      'tuple'
      'poly'
      --- A list of strings of what the string could be
      'enum'
      --- A generic found within function definitions
      'generic'
      --- A type provided from outside the teal compiler
      'named'
   end

   --- The kind of type this is
   kind: Kinds
   --- The description of the type, every type can have a description since of records
   description: string
   --- The name of the type, usually inherited from their parent
   name: string
   --- The custom name of a type
   customName: string | nil
   --- The parent of the type, this doesn't exist for top-level types
   parent: Typetype record | nil
   --- The file-system tree that needs to be followed
   tree: string
   --- If the type is hidden or not
   hidden: boolean
   --- The Y position of a type
   yPos: integer | nil
   --- The stopping point of a type
   stopY: integer | nil
   --- The source of where the type was found
   source: string | nil
   --- Exists on function types
   func: Functiontype record | nil
   --- Either a record or a module
   struct: Structtype record
   --- The key of the table
   key: Typetype record | nil
   --- The value of the table
   value: Typetype record | nil
   --- The types that form this type, eg `type1 | type2` or `{type1, type2}`
   types: {Typetype record} | nil
   --- The values of an enum, its separate from types due to enums being "complex" aka blocked
   values: {Typetype record} | nil
   --- Signify that a type has been deprecated
   deprecated: boolean | string
   --- @{hidden}
   tp: tl<invalid type>.TypeInfo
end

local record Statetype record
   report: tl<invalid type>.TypeReport
   files: {string: string}
end

-- Token isn't exposed
local record Tokentype record
   x: integer
   y: integer
   i: integer
   tk: string
   kind: string
end

-- Same with Node
local record Nodetype record
   {Nodetype record}

   y: integer
   x: integer
   tk: string
   kind: string
end

local function generateGlobalsfunction({string}): string(modules{string}: {string}): string
   local mod{string} = {'global toLoad = {'}

   for inumber = 1, #modules{string} do
      local strstring = modules{string}[inumber]
      -- globals are easier to find
      table.insert(mod{string}, '["' .. strstring .. '"] = require \'' .. strstring .. '\',')
   end

   table.insert(mod{string}, '}')

   return table.concat(mod{string}, '\n')
end

-- Replaces teals search for one which indexes the virtual modules
local function newSearchfunction({string : string}, boolean, State): function(string): string, FILE, {string}(modules{string : string}: {string: string}, virtualboolean: boolean, stateState: Statetype record): function(module: string): string, FILE, {string}
   return function(modulestring: string): string, FILE, {string}
      if modules{string : string}[modulestring] then
         local modstring = modules{string : string}[modulestring]

         local fFILE: FILE
         local filenamestring: string

         -- Virtual
         if virtualboolean then
            local namestring = os.tmpname() .. '.tl'
            do
               local tmpFILE = io.open(namestring, 'w')
               tmpFILE:write(modstring)
               tmpFILE:close()
               stateState.files[namestring] = modstring
            end
            fFILE = io.open(namestring, 'r')
            filenamestring = name
         else
            do
               local tmpFILE = io.open(modstring, 'r')
               stateState.files[modstring] = tmpFILE:read('*a')
               tmpFILE:close()
            end
            fFILE = io.open(modstring, 'r')
            filenamestring = modstring
         end

         return filename, f
      else
         return nil, nil, {'We tried everything...'}
      end
   end
end

--- Get a report on all the files
---
--- The report contains all indexed files as well as the types
---
--- If the modules being passed don't exist on disk, pass `true` to the virtual flag
---
--- If the files are Lua, pass `true` to the isLua flag
local function getReportfunction({string : string}, boolean, boolean): State(modules{string : string}: {string: string}, virtualboolean: boolean, isLuaboolean: boolean): Statetype record
   local keys{string}: {string} = {}

   for istring in pairs(modules{string : string}) do
      table.insert(keys{string}, istring)
   end

   local strstring = generateGlobalsfunction({string}): string(keys{string})

   local oldSearch<invalid type> = tl<invalid type>.search_module

   local stateState = {
      files = {}
   } as Statetype record

   tl<invalid type>.search_module = newSearchfunction({string : string}, boolean, State): function(string): string, FILE, {string}(modules{string : string}, virtualboolean, stateState)

   local result<invalid type> = tl<invalid type>.process_string(strstring, isLuaboolean)

   if #result<invalid type>.syntax_errors > 1 then
      for inumber= 1, #result<invalid type>.syntax_errors do
         loggerrecord.error(result<invalid type>.syntax_errors[inumber].msg)
      end

      os.exit(-1)
   end

   local types<invalid type> = tl<invalid type>.get_types(result<invalid type>)

   tl<invalid type>.search_module = oldSearch<invalid type>

   stateState.report = types<invalid type>

   return stateState
end

--- Extract all the indexes of generated modules
local function extractModulesfunction(State, {string : string}): {string : integer}(stateState: Statetype record, mods{string : string}: {string: string}): {string: integer}
   local collected{string : integer}: {string: integer} = {}

   local pos<invalid type> = stateState.report.globals.toLoad

   for i<invalid type>, v<invalid type> in pairs(stateState.report.types[pos<invalid type>].fields) do
      if mods{string : string}[i<invalid type>] then
         collected{string : integer}[i<invalid type>] = v<invalid type>
      end
   end

   return collected{string : integer}
end

local function getDescriptionfunction(string, integer): string(contentsstring: string, yinteger: integer): string
   local split{} = {}

   do
      local posnumber = 1

      while true do
         local nextNewlinenumber = contentsstring:find('\n', posnumber, true)

         if not nextNewlinenumber then
            break
         end

         table.insert(split{}, contentsstring:sub(posnumber, nextNewlinenumber - 1))

         posnumber = nextNewlinenumber + 1
      end
   end

   local descLines{string}: {string} = {}

   local i<invalid type> = yinteger - 1

   while true do
      local line<invalid type> = split{}[i<invalid type>]

      if i<invalid type> < 0 then
         break
      else
         i<invalid type> = i<invalid type> - 1

         if line<invalid type> and line<invalid type>:match('%s*%-%-%-%.*') then
            table.insert(descLines{string}, line<invalid type>:match('%s*%-%-%- ?(.*)'))
         else
            break
         end
      end
   end

   local reversed{} = {}

   for i<invalid type> = #descLines{string}, 1, -1 do
      table.insert(reversed{}, descLines{string}[i<invalid type>])
   end

   return table.concat(reversed{}, '\n')
end

local function getTokensAtLinefunction({Token}, integer, integer): {Token}, {Token}(tokens{Token}: {Tokentype record}, yinteger: integer, xinteger: integer): {Tokentype record}, {Tokentype record}
   --; TODO, use a binary search
   local startedboolean = false
   local tokensAtSig{} = {}
   local before{} = {}

   for inumber = 1, #tokens{Token} do
      local tokenToken = tokens{Token}[inumber]

      if tokenToken.yinteger == yinteger and tokenToken.xinteger >= xinteger then
         startedboolean = true
         table.insert(tokensAtSig{}, tokenToken)
      elseif tokenToken.yinteger ~= yinteger and startedboolean then
         break
      elseif not startedboolean then
         table.insert(before{}, tokenToken)
      end
   end

   return tokensAtSig{}, before{}
end

local function getTokensBetweenfunction({Token}, integer, integer, integer, integer): {Token}(tokens{Token}: {Tokentype record}, y1integer: integer, y2integer: integer, x1integer: integer, x2integer: integer): {Tokentype record}
   --; TODO, use a binary search
   local startedboolean = false
   local tokensBetween{} = {}

   for inumber = 1, #tokens{Token} do
      local tokenToken = tokens{Token}[inumber]

      if tokenToken.y >= y1integer and tokenToken.x >= x1integer then
         startedboolean = true
      end

      if tokenToken.y > y2integer or (tokenToken.y == y2integer and tokenToken.x >= x2integer) then
         break
      end

      if startedboolean then
         table.insert(tokensBetween{}, tokenToken)
      end
   end

   return tokensBetween{}
end

local function getStopfunction({Token}, integer, integer, boolean): Token, integer(tokens{Token}: {Tokentype record}, yinteger: integer, xinteger: integer, allowFunctionboolean: boolean): Tokentype record, integer
   local canStartboolean = false
   local startedboolean = false

   -- These are what you can use **within a record** which can open a scope
   local opensScope{string : boolean}: {string: boolean} = {
      ['enum'] = true,
      ['record'] = true,
      -- This expects a record and records can only have inline functions
      ['function'] = allowFunctionboolean
   }

   local deepnessnumber = 0
   local lastTokenToken: Tokentype record

   for inumber = 1, #tokens{Token} do
      local tokenToken = tokens{Token}[inumber]

      if tokenToken.yinteger == yinteger and tokenToken.xinteger >= xinteger then
         canStartboolean = true
      end

      if canStartboolean and opensScope{string : boolean}[tokenToken.tk] then
         startedboolean = true
      end

      if startedboolean and (tokenToken.kind == 'keyword' and tokenToken.tk == 'end') then
         deepnessnumber = deepnessnumber - 1
      elseif startedboolean and opensScope{string : boolean}[tokenToken.tk] then
         deepnessnumber = deepnessnumber + 1
      end

      if startedboolean and deepnessnumber == 0 then
         lastTokenToken = tokenToken

         return lastTokenToken, inumber
      end
   end
end

local function parseKindfunction(Node, {string : integer})(toParseNode: Nodetype record, lines{string : integer}: {string: integer})
   if toParseNode.kind == 'expression_list' then
      for inumber = 1, #toParseNode do
         parseKindfunction(Node, {string : integer})(toParseNode[inumber], lines{string : integer})
      end
   elseif toParseNode.kind == 'table_literal' then
      for inumber = 1, #toParseNode do
         lines{string : integer}[toParseNode[inumber].tk] = toParseNode[inumber].y
      end
   else
      -- `local_type` would be a record which we can already parse
      -- Records are easier to parse since it isn't the entire Lua syntax
      error('Unexpected kind: ' .. toParseNode.kind)
   end
end

local function recursiveFindfunction(Node, integer, integer, {Node : boolean}, integer): node, string(nodeNode: Nodetype record, xinteger: integer, yinteger: integer, visited{Node : boolean}: {Nodetype record: boolean}, depthinteger: integer): nodeNode, string
   visited{Node : boolean} = visited{Node : boolean} or {}
   depthinteger = depthinteger or 0

   if depthinteger > 10000 then
      return nil, 'max-depth'
   end

   if visited{Node : boolean}[nodeNode] then
      return nil, 'visited'
   end

   visited{Node : boolean}[nodeNode] = true

   local viewed{} = {}

   for _string, vNode in pairs(nodeNode as {string: Nodetype record}) do
      if type(vNode) == 'table' and vNode.xinteger and vNode.yinteger then
         table.insert(viewed{}, vNode)

         if vNode.xinteger == xinteger and vNode.yinteger == yinteger then
            return vNode
         else
            local succnode = recursiveFindfunction(Node, integer, integer, {Node : boolean}, integer): node, string(vNode, xinteger, yinteger, visited{Node : boolean}, depthinteger + 1)

            if succnode then
               return succnode
            end
         end
      end
   end

   return nil, 'viewed-all'
end

--- Create a (parsable) type from an index and a report
local function createTypefunction(integer, State, boolean): Type(indexinteger: integer, reportState: Statetype record, noNominalboolean: boolean): Typetype record
   local tp<invalid type> = reportState.reportState.types[indexinteger]

   local codes<invalid type> = tl<invalid type>.typecodes

   local conversions{integer : Type.kind}: {integer: Typetype record.kind} = {
      [codes<invalid type>.NIL] = 'nil',
      [codes<invalid type>.BOOLEAN] = 'boolean',
      [codes<invalid type>.NUMBER] = 'number',
      [codes<invalid type>.STRING] = 'string',
      [codes<invalid type>.USERDATA] = 'userdata',
      [codes<invalid type>.THREAD] = 'thread',
      [codes<invalid type>.TABLE] = 'table',
      [codes<invalid type>.ANY] = 'any',
      [codes<invalid type>.UNKNOWN] = 'unknown'
   }

   if codes<invalid type>.INTEGER then
      conversions{integer : Type.kind}[codes<invalid type>.INTEGER] = 'integer'
   end

   local outType: Typetype record

   if not tp<invalid type> then
      -- if we don't have a tp, get out of here
      return {
         kind = 'unknown'
      }
   end

   if conversions{integer : Type.kind}[tp<invalid type>.t] then
      -- Basic type
      outType = {
         kind = conversions{integer : Type.kind}[tp<invalid type>.t]
      }
   elseif tp<invalid type>.t == codes<invalid type>.FUNCTION then
      local tokens<invalid type> = tl<invalid type>.lex(reportState.files[tp<invalid type>.file])

      local paramNames{} = {}

      local startedboolean = false
      local tokensAtSig{Token} = getTokensAtLinefunction({Token}, integer, integer): {Token}, {Token}(tokens<invalid type> as {Tokentype record}, tp<invalid type>.y, tp<invalid type>.x)

      for inumber = 1, #tokensAtSig{Token} do
         local tokenToken = tokensAtSig{Token}[inumber]
         if tokenToken.tk == '(' then
            startedboolean = true
         elseif startedboolean and tokenToken.tk == ')' then
            break
         elseif startedboolean and (tokenToken.kind == 'identifier' or tokenToken.kind == '...') and (tokensAtSig{Token}[inumber - 1].kind == ',' or tokensAtSig{Token}[inumber - 1].kind == '(') then
            table.insert(paramNames{}, tokensAtSig{Token}[inumber].tk)
         end
      end

      local params{} = {}

      for inumber = 1, #tp<invalid type>.args do
         local argTypeType = createTypefunction(integer, State, boolean): Type(tp<invalid type>.args[inumber][1], reportState, noNominalboolean)

         argTypeType.name = paramNames{}[inumber]
         table.insert(params{}, argTypeType)
      end

      local rets{} = {}

      for inumber = 1, #tp<invalid type>.rets{} do
         table.insert(rets{}, createTypefunction(integer, State, boolean): Type(tp<invalid type>.rets{}[inumber][1], reportState, noNominalboolean))
      end

      --; TODO - get functions stop

      outType = {
         kind = 'function',
         description = getDescriptionfunction(string, integer): string(reportState.files[tp<invalid type>.file], tp<invalid type>.y),
         source = tp<invalid type>.file,
         yPos = tp<invalid type>.y,
         func = {
            params{} = params{},
            returns = rets{},
            vararg = tp<invalid type>.vararg
         }
      }
   elseif tp<invalid type>.t == codes<invalid type>.ARRAY then
      outType = {
         kind = 'table',
         key = {
            kind = 'number'
         },
         value = createTypefunction(integer, State, boolean): Type(tp<invalid type>.elements, reportState, noNominalboolean)
      }
   elseif tp<invalid type>.t == codes<invalid type>.RECORD and tp<invalid type>.file == nil then
      -- Files are the only "class" in Lua
      outType = {
         kind = 'named',
         customName = 'file'
      }
   elseif (tp<invalid type>.t == codes<invalid type>.RECORD or tp<invalid type>.t == codes<invalid type>.ARRAYRECORD) and tp<invalid type>.file then
      local tokens<invalid type> = tl<invalid type>.lex(reportState.files[tp<invalid type>.file])

      local tokensAtSig{Token}, before{Token} = getTokensAtLinefunction({Token}, integer, integer): {Token}, {Token}(tokens<invalid type> as {Tokentype record}, tp<invalid type>.y, tp<invalid type>.x)

      local namestring: string
      local lastSigToken = tokens<invalid type>[#tokensAtSig{Token} + #before{Token} + 1] as Tokentype record

      local isRecordboolean = false

      for inumber = 1, #tokensAtSig{Token} do
         if tokensAtSig{Token}[inumber].tk == 'record' then
            isRecordboolean = true
         end

         if tokensAtSig{Token}[inumber].kind == 'identifier' and tokensAtSig{Token}[inumber - 1] and tokensAtSig{Token}[inumber - 1].tk == 'record' then
            namestring = tokensAtSig{Token}[inumber].tk
            break
         end
      end

      if not namestring then
         -- It's a sub-record
         if before{Token}[#before{Token}].kind == 'identifier' and before{Token}[#before{Token} - 1] and before{Token}[#before{Token} - 1].tk == 'record' then
            namestring = before{Token}[#before{Token}].tk
            -- Correct the X position to include `record`
            tp<invalid type>.x = before{Token}[#before{Token} - 1].x as integer
            -- Same with the Y
            tp<invalid type>.y = before{Token}[#before{Token} - 1].y as integer

            lastSigToken = tokensAtSig{Token}[1]
         end
      end

      outType = {
         kind = 'custom',
         namestring = namestring,
         description = getDescriptionfunction(string, integer): string(reportState.files[tp<invalid type>.file], tp<invalid type>.y),
         source = tp<invalid type>.file,
         yPos = tp<invalid type>.y,
         struct = {
            fields = {},
            isRecordboolean = isRecordboolean
         }
      }

      for i<invalid type>, v<invalid type> in pairs(tp<invalid type>.fields) do
         local typType = createTypefunction(integer, State, boolean): Type(v<invalid type>, reportState, noNominalboolean)

         typType.namestring = i<invalid type>
         typType.parent = outType

         outType.struct.fields[i<invalid type>] = typType
      end

      -- This part is (more) likely to fail
      xpcall(function()
         if namestring then
            local lastTokenToken = getStopfunction({Token}, integer, integer, boolean): Token, integer(tokens<invalid type> as {Tokentype record}, tp<invalid type>.y, tp<invalid type>.x)

            outType.stopY = lastTokenToken.y

            local tokensBetween{Token} = getTokensBetweenfunction({Token}, integer, integer, integer, integer): {Token}(tokens<invalid type> as {Tokentype record}, lastSigToken.y, lastTokenToken.y, lastSigToken.x, lastTokenToken.x)

            local inumber = 1

            while inumber <= #tokensBetween{Token} do
               if tokensBetween{Token}[inumber].kind == 'identifier' and (tokensBetween{Token}[inumber].tk == 'record' or tokensBetween{Token}[inumber].tk == 'enum') and
                  tokensBetween{Token}[inumber + 1] and tokensBetween{Token}[inumber + 1].kind == 'identifier'
               then
                  local _Token, posinteger = getStopfunction({Token}, integer, integer, boolean): Token, integer(tokensBetween{Token}, tokensBetween{Token}[inumber].y, tokensBetween{Token}[inumber].x)

                  -- They already get documented since they are records

                  inumber = posinteger + 1
               else
                  local fieldTokenToken = tokensBetween{Token}[inumber]

                  if fieldTokenToken.tk == 'metamethod' then
                     inumber = inumber + 1

                     outType.struct.fields[tokensBetween{Token}[inumber].tk] = {
                        kind = 'function',
                        source = tp<invalid type>.file,
                        namestring = tokensBetween{Token}[inumber].tk,
                        yPos = fieldTokenToken.y,
                        func = { --; TODO - Extract these
                                 --; Teal doesn't expose metamethods within requires
                           params = {},
                           returns = {},
                           vararg = {},
                           isMetamethod = true
                        } as Functiontype record
                     }
                  else
                     local foundboolean = false

                     local outFieldType = outType.struct.fields[fieldTokenToken.tk]

                     assert(outFieldType, 'The parser seems to have parsed the wrong fields. Collected field name: ' .. fieldTokenToken.tk)

                     outFieldType.description = getDescriptionfunction(string, integer): string(reportState.files[tp<invalid type>.file], fieldTokenToken.y)

                     local stacknumber = 0
                     local ignoreNextboolean = outFieldType.kind == 'function'
                     -- logger.debug(atLine)

                     for knumber = inumber + 2, #tokensBetween{Token} do
                        -- We need to watchout for composition types eg functions, tables

                        local tokenToken = tokensBetween{Token}[knumber]

                        if tokenToken.kind == '{' or tokenToken.kind == '(' then
                           stacknumber = stacknumber + 1
                        elseif tokenToken.kind == '}' or tokenToken.kind == ')' then
                           stacknumber = stacknumber - 1
                        end

                        if tokenToken.kind == ':' and tokenToken.tk == ':' and stacknumber == 0 and not ignoreNextboolean then
                           inumber = knumber - 1

                           foundboolean = true

                           break
                        elseif ignoreNextboolean and tokenToken.kind == ':' and tokenToken.tk == ':' then
                           ignoreNextboolean = false
                        end
                     end

                     if not foundboolean then
                        break
                     end
                  end
               end
            end
         else
            local _<invalid type>, node<invalid type> = tl<invalid type>.parse_program(tokens<invalid type>)

            local collectedNodenode, reasonstring = recursiveFindfunction(Node, integer, integer, {Node : boolean}, integer): node, string(node<invalid type> as Nodetype record, tp<invalid type>.x, tp<invalid type>.y)

            if not collectedNodenode then
               error('Unable to find node: ' .. tostring(reasonstring))
            end

            local lines{} = {}

            parseKindfunction(Node, {string : integer})(collectedNodenode, lines{})

            for istring, vinteger in pairs(lines{}) do
               outType.struct.fields[istring].description = getDescriptionfunction(string, integer): string(reportState.files[tp<invalid type>.file], vinteger)
            end
         end
      end, function(err<unknown type>)
         loggerrecord.error(string.format(
            'Parsing error with field comments!\n' ..
            '  Error: %s\n' ..
            '  Report this to https://github.com/SovietKitsune/tealdoc/issues\n' ..
            '  Recovering... (Field descriptions might not exist)\n\n%s',
            err<unknown type>,
            debug.traceback()
         ))
      end)
   elseif tp<invalid type>.t == codes<invalid type>.MAP then
      outType = {
         kind = 'table',
         key = createTypefunction(integer, State, boolean): Type(tp<invalid type>.keys, reportState, noNominalboolean),
         value = createTypefunction(integer, State, boolean): Type(tp<invalid type>.values, reportState, noNominalboolean)
      }
   elseif tp<invalid type>.t == codes<invalid type>.TUPLE or tp<invalid type>.t == codes<invalid type>.IS_UNION or tp<invalid type>.t == codes<invalid type>.IS_POLY then
      local types{} = {}

      for inumber = 1, #tp<invalid type>.types{} do
         local parsedType = createTypefunction(integer, State, boolean): Type(tp<invalid type>.types{}[inumber], reportState, noNominalboolean)

         table.insert(types{}, parsedType)
      end

      outType = {
         kind =
            tp<invalid type>.t == codes<invalid type>.TUPLE and 'tuple' or
            tp<invalid type>.t == codes<invalid type>.IS_UNION and 'union' or
            tp<invalid type>.t == codes<invalid type>.IS_POLY and 'poly',
         types{} = types{}
      }
   elseif tp<invalid type>.t == codes<invalid type>.NOMINAL and not noNominalboolean then
      -- Reference to another record
      -- As long as its not recursion we *should* be fine

      -- We don't set the parent on this one since it would be mostly pointless
      local refType = createTypefunction(integer, State, boolean): Type(tp<invalid type>.refType, reportState, true)

      -- This caused too many issues
      outType = ref
   elseif tp<invalid type>.t == codes<invalid type>.ENUM then
      local tokens<invalid type> = tl<invalid type>.lex(reportState.files[tp<invalid type>.file])

      local tokensAtSig{Token}, before{Token} = getTokensAtLinefunction({Token}, integer, integer): {Token}, {Token}(tokens<invalid type> as {Tokentype record}, tp<invalid type>.y, tp<invalid type>.x)

      local namestring: string
      local lastSigToken = tokens<invalid type>[#tokensAtSig{Token} + #before{Token} + 1] as Tokentype record

      for inumber = 1, #tokensAtSig{Token} do
         if tokensAtSig{Token}[inumber].kind == 'identifier' and tokensAtSig{Token}[inumber - 1] and tokensAtSig{Token}[inumber - 1].tk == 'enum' then
            namestring = tokensAtSig{Token}[inumber].tk
            break
         end
      end

      if not namestring then
         -- It's a sub-record
         if before{Token}[#before{Token}].kind == 'identifier' and before{Token}[#before{Token} - 1] and before{Token}[#before{Token} - 1].tk == 'enum' then
            namestring = before{Token}[#before{Token}].tk
            -- Correct the X position to include `record`
            tp<invalid type>.x = before{Token}[#before{Token} - 1].x
            -- Same with the Y
            tp<invalid type>.y = before{Token}[#before{Token} - 1].y

            lastSigToken = tokensAtSig{Token}[1]
         end
      end

      local lastTokenToken = getStopfunction({Token}, integer, integer, boolean): Token, integer(tokens<invalid type> as {Tokentype record}, tp<invalid type>.y, tp<invalid type>.x)

      local between{Token} = getTokensBetweenfunction({Token}, integer, integer, integer, integer): {Token}(tokens<invalid type> as {Tokentype record}, lastSigToken.y, lastTokenToken.y, lastSigToken.x, lastTokenToken.x)

      local values{} = {}

      for inumber = 1, #between{Token} do
         local tokenToken = between{Token}[inumber]

         table.insert(values{}, {
            kind = 'string',
            namestring = tokenToken.tk,
            yPos = tokenToken.y,
            stopY = tokenToken.y,
            description = getDescriptionfunction(string, integer): string(reportState.files[tp<invalid type>.file], tokenToken.y)
         })
      end

      outType = {
         kind = 'enum',
         values{} = values{},
         namestring = namestring,
