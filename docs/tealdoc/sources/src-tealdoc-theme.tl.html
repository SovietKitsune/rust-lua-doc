local record Themetype record
   record Code
      text: string
      background: string
      border: string
      syntax: Syntax
   end

   record Syntax
      comment: string
      constant: string
      identifier: string
      keyword: string
      number: string
      string: string
      library: string
   end

   text: string
   background: string
   tooltip: string
   outline: string
   outlineMain: string
   outlineBackground: string

   link: string
   nav: string
   navLink: string
   code: Code
end

local supportedTokens{string : boolean}: {string: boolean} = {
   comment = true,
   constant = true,
   identifier = true,
   keyword = true,
   number = true,
   string = true,
   library = true
}

local function createCSSfunction(): string(): string
   local sheet{string} = {
      [[
         .tooltip-parent .tooltip {
            visibility: hidden;
            text-align: center;

            padding-left: 0.25rem;
            padding-right: 0.25rem;

            border-radius: 1px;
            position: absolute;
            z-index: 1;
         }

         .tooltip-parent:hover .tooltip {
            visibility: visible;
         }
      ]]
   }

   for themestring, stylesTheme in pairs(themes{string : Theme}) do
      table.insert(sheet{string}, string.format('/* %s theme */', themestring))
      table.insert(
         sheet{string},
         string.format([[
            .%s {
               background-color: %s;
               color: %s;
            }

            .%s .border-black {
               border-color: %s;
            }

            .%s .top-border {
               background-color: %s;
               border-color: %s;
            }

            .%s a {
               color: %s;
            }

            .%s nav {
               background-color: %s;
            }

            .%s nav a {
               color: %s;
            }

            .%s .tippy-box[data-theme~='auto'] {
               background-color: %s;
            }

            .%s .tooltip-parent .tooltip {
               background-color: %s;
               color: %s;
            }

            .%s .code-block {
               background-color: %s;
               border-color: %s;
            }
         ]],
         themestring,
         stylesTheme.background,
         stylesTheme.text,
         themestring,
         stylesTheme.outline,
         themestring,
         stylesTheme.outlineBackground,
         stylesTheme.outlineMain,
         themestring,
         stylesTheme.link,
         themestring,
         stylesTheme.nav,
         themestring,
         stylesTheme.navLink,
         themestring,
         stylesTheme.tooltip,
         themestring,
         stylesTheme.code.border,
         stylesTheme.code.text,
         themestring,
         stylesTheme.code.background,
         stylesTheme.code.border
      ))

      for kindstring, stylestring in pairs(stylesTheme.code.syntax as {string: string}) do
         table.insert(sheet{string}, string.format('.%s .code-%s { color: %s; }', themestring, kindstring, stylestring))
      end

      table.insert(sheet{string}, string.format('.%s .code-text { color: %s; }', themestring, stylesTheme.code.text))
   end

   return table.concat(sheet{string}, '\n')
end

local function isNonTextfunction(string): boolean(kindstring: string): boolean
   return supportedTokens{string : boolean}[kindstring]
end

return {
   themes{string : Theme} = themes{string : Theme},
   createCSSfunction(): string = createCSSfunction(): string,
   css = createCSSfunction(): string(), -- Generate CSS at load, if a new theme is added, it can manually be regenerated
   isNonTextfunction(string): boolean = isNonTextfunction(string): boolean
}
