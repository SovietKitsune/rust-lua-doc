--- The core functions of math, addition, subtraction, multiplication, and division
local corerecord = {
   --- This is a fun variable
   isFun = true,
   --- Add two numbers together
   ---
   --- ```lua
   --- core.add(1, 2) -- 3
   --- ```
   add = function(anumber: number, bnumber: number): number
      return anumber + bnumber
   end,
   --- Subtract two numbers apart
   ---
   --- ```lua
   --- core.subtract(1, 2) -- -1
   --- ```
   sub = function(anumber: number, bnumber: number): number
      return anumber - bnumber
   end,
   --- Multiply two numbers together
   ---
   --- ```lua
   --- core.multiply(1, 2) -- 2
   --- ```
   mul = function(anumber: number, bnumber: number): number
      return anumber * bnumber
   end,
   --- Divide two numbers apart
   ---
   --- ```lua
   --- core.divide(1, 2) -- 0.5
   --- ```
   div = function(anumber: number, bnumber: number): number
      return anumber / bnumber
   end
}

return corerecord
